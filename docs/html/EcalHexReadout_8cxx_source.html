36  Double_t numberOfHexagonsInAColumn;

40  xloop = xstart;
41  yloop = ystart + a * sqrt_three / 2.0;
42

44  ytemp = yloop; // Resets the temp variable
45 
46  // Determine the number of hexagons in that column
47  if (sCounter % 2 == 0) {
48  numberOfHexagonsInAColumn = k;
49  } else {
50  numberOfHexagonsInAColumn = k - 1;
51  }
52 
53  for (int kCounter = 0; kCounter < numberOfHexagonsInAColumn; kCounter++) {
54

56  x[0] = xloop;
57  y[0] = ytemp;
58  x[1] = x[0] + a / 2.0;
59  y[1] = y[0] + a * sqrt_three / 2.0;
60  x[2] = x[1] + a;

63  y[3] = y[1] - a * sqrt_three / 2.0;
64  x[4] = x[2];

66  x[5] = x[1];
