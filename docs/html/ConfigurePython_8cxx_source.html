196 
197  for (auto lib : libraries_) {
198  EventProcessorFactory::getInstance().loadLibrary(lib);
199  }
200 
201  for (auto proc : sequence_) {

203  if (ep == 0) {
204  EXCEPTION_RAISE("UnableToCreate", "Unable to create instance '" + proc.instancename_ + "' of class '" + proc.classname_ + "'");
205  }
206  ep->configure(proc.params_);
207  p->addToSequence(ep);
208  }
209  for (auto file : inputFiles_) {
210  p->addFileToProcess(file);
211  }
212  for (auto file : outputFiles_) {
213  p->addOutputFileName(file);
214  }
215 
216  for (auto rule : keepRules_) {
217  p->addDropKeepRule(rule);
218  }
219  if (run_ > 0)
220  p->setRunNumber(run_);
221  p->setEventLimit(eventLimit_);
222
