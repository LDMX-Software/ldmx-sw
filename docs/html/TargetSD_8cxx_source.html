42  G4ParticleDefinition* pdef = aStep->GetTrack()->GetDefinition();

44  if (pdef == G4Geantino::Definition() || pdef == G4ChargedGeantino::Definition()) {
45  isGeantino = true;
46  }
47 
48  // Get the edep from the step.

50 
51  // Skip steps with no energy dep which come from non-Geantino particles.
52  if (edep == 0.0 && !isGeantino) {
53  if (verboseLevel > 2) {
54  std::cout << "CalorimeterSD skipping step with zero edep."
55  << std::endl << std::endl;
56  }
57  return false;
58  }
59 
60  // Create a new hit object using the ROOT event.
61  G4CalorimeterHit* hit = new G4CalorimeterHit();
62 
63  // Set the edep.
64  hit->setEdep(edep);
65 
66  // Set the position.
67  G4StepPoint* prePoint = aStep->GetPreStepPoint();
68  G4StepPoint* postPoint = aStep->GetPostStepPoint();
69  G4ThreeVector position = 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
70  G4ThreeVector volumePosition = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()
71  ->GetTopTransform().Inverse().TransformPoint(G4ThreeVector());
72  hit->setPosition(position[0], position[1], volumePosition.z());
73 
74  // Set the global time.
75  hit->setTime(aStep->GetTrack()->GetGlobalTime());
76 
77  // Set the ID on the hit.
78  int subdet = prePoint->GetTouchableHandle()->GetHistory()->GetVolume(1)->GetCopyNo();
79  detID_->setFieldValue(0, subdet);
80  hit->setID(detID_->pack());
81 
82  // Set the track ID on the hit.
83  hit->setTrackID(aStep->GetTrack()->GetTrackID());
84 
85  if (this->verboseLevel > 2) {
86  std::cout << "Created new SimCalorimeterHit in detector "
87  << this->GetName() << " with subdet ID " << subdet
88  << std::endl;
89  hit->Print();
90  std::cout << std::endl;
91  }
92

94
