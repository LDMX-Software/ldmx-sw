144

147  G4String auxUnit = iaux->unit;

149  G4LogicalVolume* lv = (*lvciter);

172  if (magField != NULL) {

174  lv->SetFieldManager(mgr,

176  std::cout << "Assigned magnetic field " << magFieldName

185  } else if (auxType == "Region") {
186  G4String regionName = auxVal;
187  G4Region* region = G4RegionStore::GetInstance()->GetRegion(
188  regionName);
189  if (region != NULL) {
190  region->AddRootLogicalVolume(lv);

193  } else {

199  } else if (auxType == "VisAttributes") {
200  G4String visName = auxVal;
201  G4VisAttributes* visAttributes =
202  VisAttributesStore::getInstance()->getVisAttributes(
203  visName);
204  if (visAttributes != NULL) {
205  lv->SetVisAttributes(visAttributes);

220 void AuxInfoReader::createMagneticField(G4String magFieldName,
221  const G4GDMLAuxListType* auxInfoList) {
222 
223  // Find type of the mag field.
224  G4String magFieldType("");
225  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
226  auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
227 
228  G4String auxType = iaux->type;
229  G4String auxVal = iaux->value;
230 
231  if (auxType == "MagneticFieldType") {
232  magFieldType = auxVal;
233  break;
234  }
235  }
236 
237  if (magFieldType == "") {
238  G4Exception("", "", FatalException,
239  "Missing MagFieldType for magnetic field definition.");
240  }
241 
242  G4MagneticField* magField = NULL;
243 
244  // Create a uniform mag field using the built-in Geant4 type.
245  if (magFieldType == "G4UniformMagField") {
246  string::size_type sz;
247  double bx, by, bz;
248  bx = by = bz = 0.;
249  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
250  auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
251 
252  G4String auxType = iaux->type;
253  G4String auxVal = iaux->value;
254  G4String auxUnit = iaux->unit;
255 
256  G4String expr = auxVal + "*" + auxUnit;
257  if (auxType == "bx") {
258  bx = eval_->Evaluate(expr);
259  } else if (auxType == "by") {
260  by = eval_->Evaluate(expr);
261  } else if (auxType == "bz") {
262  bz = eval_->Evaluate(expr);
263  }
264  }
265  G4ThreeVector fieldComponents(bx, by, bz);
266  magField = new G4UniformMagField(fieldComponents);
267 
268  std::cout << "Created G4UniformMagField " << magFieldName
269  << " with field components " << fieldComponents << std::endl
270  << std::endl;
271 
272  // Create a global 3D field map by reading from a data file.
273  } else if (magFieldType == "MagneticFieldMap3D") {
274 
275  string fileName;
276  double offsetX, offsetY, offsetZ;
277 
278  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
279  auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
280 
281  G4String auxType = iaux->type;
282  G4String auxVal = iaux->value;
283  G4String auxUnit = iaux->unit;
284 
285  G4String expr = auxVal + "*" + auxUnit;
286 
287  if (auxType == "File") {
288  fileName = auxVal;
289  } else if (auxType == "OffsetX") {
290  offsetX = eval_->Evaluate(expr);
291  } else if (auxType == "OffsetY") {
292  offsetY = eval_->Evaluate(expr);
293  } else if (auxType == "OffsetZ") {
294  offsetZ = eval_->Evaluate(expr);
295  }
296  }
297 
298  if (fileName.size() == 0) {
299  G4Exception("", "", FatalException,
300  "File info with field data was not provided.");
301  }
302 
303  // Create new 3D field map.
304  G4MagneticField* fieldMap = new MagneticFieldMap3D(fileName.c_str(),
305  offsetX, offsetY, offsetZ);
306 
307  // Assign field map as global field.
308  G4FieldManager* fieldMgr =
309  G4TransportationManager::GetTransportationManager()->GetFieldManager();
310  if (fieldMgr->GetDetectorField() != nullptr) {
311  G4Exception("", "", FatalException,
312  "Global mag field was already assigned.");
313  }
314  fieldMgr->SetDetectorField(fieldMap);
315  fieldMgr->CreateChordFinder(fieldMap);
316 
317  } else {
318  std::cerr << "Unknown MagFieldType in auxiliary info: " << magFieldType
319  << std::endl;
320  G4Exception("", "", FatalException,
321  "Unknown MagFieldType in auxiliary info.");
322  }
323 
324  MagneticFieldStore::getInstance()->addMagneticField(magFieldName, magField);
325 }
