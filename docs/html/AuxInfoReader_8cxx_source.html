322 void AuxInfoReader::createMagneticField(G4String magFieldName,
323  const G4GDMLAuxListType* auxInfoList) {
324 
325  // Find type of the mag field.
326  G4String magFieldType("");
327  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
328  auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
329 
330  G4String auxType = iaux->type;
331  G4String auxVal = iaux->value;
332 
333  if (auxType == "MagneticFieldType") {
334  magFieldType = auxVal;
335  break;
336  }
337  }
338 
339  if (magFieldType == "") {
340  G4Exception("", "", FatalException,
341  "Missing MagFieldType for magnetic field definition.");
342  }
343 
344  G4MagneticField* magField = NULL;
345 
346  // Create a uniform mag field using the built-in Geant4 type.
347  if (magFieldType == "G4UniformMagField") {
348  string::size_type sz;
349  double bx, by, bz;
350  bx = by = bz = 0.;
351  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
352  auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
353 
354  G4String auxType = iaux->type;
355  G4String auxVal = iaux->value;
356  G4String auxUnit = iaux->unit;
357 
358  G4String expr = auxVal + "*" + auxUnit;
359  if (auxType == "bx") {
360  bx = eval_->Evaluate(expr);
361  } else if (auxType == "by") {
362  by = eval_->Evaluate(expr);
363  } else if (auxType == "bz") {
364  bz = eval_->Evaluate(expr);
365  }
366  }
367  G4ThreeVector fieldComponents(bx, by, bz);
368  magField = new G4UniformMagField(fieldComponents);
369 
370  std::cout << "Created G4UniformMagField " << magFieldName
371  << " with field components " << fieldComponents << std::endl
372  << std::endl;
373 
374  // Create a global 3D field map by reading from a data file.
375  } else if (magFieldType == "MagneticFieldMap3D") {
376 
377  string fileName;
378  double offsetX, offsetY, offsetZ;
379 
380  for (std::vector<G4GDMLAuxStructType>::const_iterator iaux =
381  auxInfoList->begin(); iaux != auxInfoList->end(); iaux++) {
382 
383  G4String auxType = iaux->type;
384  G4String auxVal = iaux->value;
385  G4String auxUnit = iaux->unit;
386 
387  G4String expr = auxVal + "*" + auxUnit;
388 
389  if (auxType == "File") {
390  fileName = auxVal;
391  } else if (auxType == "OffsetX") {
392  offsetX = eval_->Evaluate(expr);
393  } else if (auxType == "OffsetY") {
394  offsetY = eval_->Evaluate(expr);
395  } else if (auxType == "OffsetZ") {
396  offsetZ = eval_->Evaluate(expr);
397  }
398  }
399 
400  if (fileName.size() == 0) {
401  G4Exception("", "", FatalException,
402  "File info with field data was not provided.");
403  }
404 
405  // Create new 3D field map.
406  G4MagneticField* fieldMap = new MagneticFieldMap3D(fileName.c_str(),
407  offsetX, offsetY, offsetZ);
408 
409  // Assign field map as global field.
410  G4FieldManager* fieldMgr =
411  G4TransportationManager::GetTransportationManager()->GetFieldManager();
412  if (fieldMgr->GetDetectorField() != nullptr) {
413  G4Exception("", "", FatalException,
414  "Global mag field was already assigned.");
415  }
416  fieldMgr->SetDetectorField(fieldMap);
417  fieldMgr->CreateChordFinder(fieldMap);
418 
419  } else {
420  std::cerr << "Unknown MagFieldType in auxiliary info: " << magFieldType
421  << std::endl;
422  G4Exception("", "", FatalException,
423  "Unknown MagFieldType in auxiliary info.");
424  }
425 
426  MagneticFieldStore::getInstance()->addMagneticField(magFieldName, magField);
427 }
