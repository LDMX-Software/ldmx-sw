151 void RootPersistencyManager::writeHitsCollections(const G4Event* anEvent, Event* outputEvent) {
152 
153  // Clear the hits from last event.
154  for (auto entry : outputHitsCollections_) {
155  entry.second->Clear("C");
156  }
157 
158  // Get the HC of this event.
159  G4HCofThisEvent* hce = anEvent->GetHCofThisEvent();
160  int nColl = hce->GetNumberOfCollections();
161 
162  // Loop over all hits collections.
163  for (int iColl = 0; iColl < nColl; iColl++) {
164 
165  // Get a hits collection and its name.
166  G4VHitsCollection* hc = hce->GetHC(iColl);
167  std::string collName = hc->GetName();
168 
169  // Get the target output collection.
170  TClonesArray* outputHitsColl = outputHitsCollections_[collName];
171 
172  // If the collection is not found in the output ROOT event, then a fatal error occurs!
173  if (!outputHitsColl) {
174  std::cerr << "ERROR: The output collection " << collName << " was not found!" << std::endl;
175  G4Exception("RootPersistencyManager::writeHitsCollections",
176  "",
177  FatalException,
178  "The output collection was not found.");
179  }
180 
181  if (dynamic_cast<G4TrackerHitsCollection*>(hc) != nullptr) {
182 
183  // Write G4TrackerHit collection to output SimTrackerHit collection.
184  G4TrackerHitsCollection* trackerHitsColl = dynamic_cast<G4TrackerHitsCollection*>(hc);
185  writeTrackerHitsCollection(trackerHitsColl, outputHitsColl);
186 
187  } else if (dynamic_cast<G4CalorimeterHitsCollection*>(hc) != nullptr) {
188 
189  G4CalorimeterHitsCollection* calHitsColl = dynamic_cast<G4CalorimeterHitsCollection*>(hc);
190  if (collName == EventConstants::ECAL_SIM_HITS) {
191  // Write ECal G4CalorimeterHit collection to output SimCalorimeterHit collection using helper class.
192  ecalHitIO_->writeHitsCollection(calHitsColl, outputHitsColl);
193  } else {
194  // Write generic G4CalorimeterHit collection to output SimCalorimeterHit collection.
195  writeCalorimeterHitsCollection(calHitsColl, outputHitsColl);
196  }
197  }
198 
199  // Add hits collection to output event.
200  outputEvent->add(collName, outputHitsColl);
201 
202  if (m_verbose > 1) {
203  std::cout << "[ RootPersistencyManager ] : Wrote " << outputHitsColl->GetEntriesFast() << " hits into " << collName << std::endl;
204  }
205  }
206 }

276 void RootPersistencyManager::setupHitsCollectionMap() {
277 
278  // Clear any state from the last run.
279  if (outputHitsCollections_.size()) {
280  for (auto entry : outputHitsCollections_) {
281  delete entry.second;
282  }
283  }
284  outputHitsCollections_.clear();
285 
286  // Create an output TClonesArray in the map for each registered HC.
287  G4SDManager* sdMgr = G4SDManager::GetSDMpointer();
288  G4HCtable* hcTable = sdMgr->GetHCtable();
289  int entries = hcTable->entries();
290  for (int i = 0; i < entries; i++) {
291  std::string sdName = hcTable->GetSDname(i);
292  std::string hcName = hcTable->GetHCname(i);
293  G4VSensitiveDetector* sd = sdMgr->FindSensitiveDetector(sdName);
294  if (dynamic_cast<CalorimeterSD*>(sd)) {
295  outputHitsCollections_[hcName] = new TClonesArray(EventConstants::SIM_CALORIMETER_HIT.c_str(), 500);
296  if (m_verbose > 1) {
297  std::cout << "[ RootPersistencyManager ] - Created SimCalorimeterHit HC " << hcName << std::endl;
298  }
299  } else if (dynamic_cast<TrackerSD*>(sd)) {
300  outputHitsCollections_[hcName] = new TClonesArray(EventConstants::SIM_TRACKER_HIT.c_str(), 50);
301  if (m_verbose > 1) {
302  std::cout << "[ RootPersistencyManager ] - Created SimTrackerHit HC " << hcName << std::endl;
303  }
304  }
305  }
