149 void RootPersistencyManager::writeHitsCollections(const G4Event* anEvent, Event* outputEvent) {
150 
151  // Clear the hits from last event.
152  for (auto entry : outputHitsCollections_) {
153  entry.second->Clear("C");
154  }
155 
156  // Get the HC of this event.
157  G4HCofThisEvent* hce = anEvent->GetHCofThisEvent();
158  int nColl = hce->GetNumberOfCollections();
159 
160  // Loop over all hits collections.
161  for (int iColl = 0; iColl < nColl; iColl++) {
162 
163  // Get a hits collection and its name.
164  G4VHitsCollection* hc = hce->GetHC(iColl);
165  std::string collName = hc->GetName();
166 
167  // Get the target output collection.
168  TClonesArray* outputHitsColl = outputHitsCollections_[collName];
169 
170  // If the collection is not found in the output ROOT event, then a fatal error occurs!
171  if (!outputHitsColl) {
172  std::cerr << "ERROR: The output collection " << collName << " was not found!" << std::endl;
173  G4Exception("RootPersistencyManager::writeHitsCollections",
174  "",
175  FatalException,
176  "The output collection was not found.");
177  }
178 
179  if (dynamic_cast<G4TrackerHitsCollection*>(hc) != nullptr) {
180 
181  // Write G4TrackerHit collection to output SimTrackerHit collection.
182  G4TrackerHitsCollection* trackerHitsColl = dynamic_cast<G4TrackerHitsCollection*>(hc);
183  writeTrackerHitsCollection(trackerHitsColl, outputHitsColl);
184 
185  } else if (dynamic_cast<G4CalorimeterHitsCollection*>(hc) != nullptr) {
186 
187  G4CalorimeterHitsCollection* calHitsColl = dynamic_cast<G4CalorimeterHitsCollection*>(hc);
188  if (collName == EventConstants::ECAL_SIM_HITS) {
189  // Write ECal G4CalorimeterHit collection to output SimCalorimeterHit collection using helper class.
190  ecalHitIO_->writeHitsCollection(calHitsColl, outputHitsColl);
191  } else {
192  // Write generic G4CalorimeterHit collection to output SimCalorimeterHit collection.
193  writeCalorimeterHitsCollection(calHitsColl, outputHitsColl);
194  }
195  }
196 
197  // Add hits collection to output event.
198  outputEvent->add(collName, outputHitsColl);
199 
200  if (m_verbose > 1) {
201  std::cout << "[ RootPersistencyManager ] : Wrote " << outputHitsColl->GetEntriesFast() << " hits into " << collName << std::endl;
202  }
203  }
204 }

274 void RootPersistencyManager::setupHitsCollectionMap() {

284  // Create an output TClonesArray in the map for each registered HC.

286  G4HCtable* hcTable = sdMgr->GetHCtable();
287  int entries = hcTable->entries();
288  for (int i = 0; i < entries; i++) {
289  std::string sdName = hcTable->GetSDname(i);
290  std::string hcName = hcTable->GetHCname(i);
291  G4VSensitiveDetector* sd = sdMgr->FindSensitiveDetector(sdName);
292  if (dynamic_cast<CalorimeterSD*>(sd)) {
293  outputHitsCollections_[hcName] = new TClonesArray(EventConstants::SIM_CALORIMETER_HIT.c_str(), 500);
294  if (m_verbose > 1) {
295  std::cout << "[ RootPersistencyManager ] - Created SimCalorimeterHit HC " << hcName << std::endl;
296  }
297  } else if (dynamic_cast<TrackerSD*>(sd)) {
298  outputHitsCollections_[hcName] = new TClonesArray(EventConstants::SIM_TRACKER_HIT.c_str(), 50);
299  if (m_verbose > 1) {
300  std::cout << "[ RootPersistencyManager ] - Created SimTrackerHit HC " << hcName << std::endl;
301  }
302  }
