110 const TObject* EventImpl::getReal(const std::string& collectionName, const std::string& passName, bool mustExist) {
111 
112  std::string branchName;
113  if (collectionName == EventConstants::EVENT_HEADER)
114  branchName = collectionName;
115  else
116  branchName = makeBranchName(collectionName, passName);
117 
118  if (passName.empty() && collectionName != EventConstants::EVENT_HEADER) {
119  auto ptr = knownLookups_.find(collectionName);
120  if (ptr != knownLookups_.end())
121  branchName = ptr->second;
122  else {
123  std::vector<std::vector<std::string>::const_iterator> matches;
124  branchName = collectionName + "_";
125  for (std::vector<std::string>::const_iterator ptr = branchNames_.begin(); ptr != branchNames_.end(); ptr++) {
126  if (!ptr->compare(0, branchName.size(), branchName))
127  matches.push_back(ptr);
128  }
129  if (matches.empty()) {
130  if (!mustExist)
131  return nullptr;
132  EXCEPTION_RAISE("ProductNotFound", "No product found for name '" + collectionName + "'");
133  } else if (matches.size() > 1) {
134  std::string names;
135  for (auto strs : matches) {
136  if (!names.empty())
137  names += ", ";
138  names += *strs;
139  }
140  if (!mustExist)
141  return nullptr;
142  EXCEPTION_RAISE("ProductAmbiguous", "Multiple products found for name '" + collectionName + "' without specified pass name (" + names + ")");
143  } else {
144  branchName = *matches.front();
145  knownLookups_[collectionName] = branchName;
146  }
147  }
148  }
149 
150  // check the objects map
151  std::map<std::string, TObject*>::const_iterator ito = objects_.find(branchName);
152  if (ito != objects_.end()) {
153  return ito->second;
154  } else if (inputTree_ == 0) {
155  EXCEPTION_RAISE("ProductNotFound", "No product found for name '" + collectionName + "' and pass '" + passName_ + "'");
156  }
157 
158  // find the active branch and update if necessary
159  std::map<std::string, TBranch*>::const_iterator itb = branches_.find(branchName);
160  if (itb != branches_.end()) {
161 
162  std::map<std::string, TObject*>::iterator ito = objects_.find(branchName);
163 
164  // update buffers if needed
165  if (itb->second->GetReadEntry() != ientry_) {
166 
167  TBranchElement* tbe = dynamic_cast<TBranchElement*>(itb->second);
168  if (!tbe)
169  itb->second->SetAddress(ito->second);
170  int nr = itb->second->GetEntry(ientry_, 1);
171  }
172 
173  // check the objects map
174 
175  if (ito != objects_.end())
176  return ito->second;
177 
178  // this case is hard to achieve
179  return 0;
180  } else {
181 
182  // ok, maybe we've not loaded this yet, look for a branch
183  TBranch* branch = inputTree_->GetBranch(branchName.c_str());
184  if (branch == 0) {
185  EXCEPTION_RAISE("ProductNotFound", "No product found for name '" + collectionName + "' and pass '" + passName_ + "'");
186  }
187 
188  // ooh, new branch!
189  TObject* top(0);
190  branch->SetAutoDelete(false);
191  branch->SetStatus(1);
192  branch->GetEntry(ientry_);
193  TBranchElement* tbe = dynamic_cast<TBranchElement*>(branch);
194  if (tbe) {
195  top = (TObject*) tbe->GetObject();
196  } else {
197  branch->SetAddress(&top);
198  }
199 
200  branches_.insert(std::pair<std::string, TBranch*>(branchName, branch));
201  objects_.insert(std::pair<std::string, TObject*>(branchName, top));
202 
203  return top;
204  }
