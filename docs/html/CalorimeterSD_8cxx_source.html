62  }
63 
64  // Create a new hit object using the ROOT event.

67  G4CalorimeterHit* hit = new G4CalorimeterHit(/*simCalorimeterHit*/);
68 
69  // Set the edep.

71 
72  // Set the position.
73  G4StepPoint* prePoint = aStep->GetPreStepPoint();
74  G4StepPoint* postPoint = aStep->GetPostStepPoint();
75  G4ThreeVector position = 0.5 * (prePoint->GetPosition() + postPoint->GetPosition());
76  G4ThreeVector volumePosition = aStep->GetPreStepPoint()->GetTouchableHandle()->GetHistory()
77  ->GetTopTransform().Inverse().TransformPoint(G4ThreeVector());
